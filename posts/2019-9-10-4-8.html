There should be usable way to implement Configurator-s for Consumer<T> instances... 

      
        public interface Configurator
        {
          void configure(object value);
        }

        public class StateToolsConsumerConfigurator : Configurator
        {
          public void configure(object value)
          {
            Consumer<StateTools> consumer = value as Consumer<StateTools>;

            if(consumer == null)
              return;
            
            consumer.set(this.stateTools);
          }
        }
